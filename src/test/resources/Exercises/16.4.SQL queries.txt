SELECT I.ID, I.SUMMARY, IL.NAME
FROM ISSUES I
         JOIN ISSUESLISTS IL ON I.ID = IL.ID;


SELECT I.ID, I.SUMMARY, U.FIRSTNAME, U.LASTNAME
FROM ISSUES I
         JOIN USERS U ON I.USER_ID_ASSIGNEDTO = U.ID
ORDER BY I.ID;


SELECT USER_ID_ASSIGNEDTO, U.FIRSTNAME, U.LASTNAME, COUNT(*) AS NUMBER_OF_ISSUES
FROM ISSUES I
         JOIN USERS U ON I.USER_ID_ASSIGNEDTO = U.ID
GROUP BY USER_ID_ASSIGNEDTO;
